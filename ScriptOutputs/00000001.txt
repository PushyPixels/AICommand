

using UnityEngine;

public class RandomizeColorScript : MonoBehaviour
{
    [MenuItem("Edit/Do Task")]
    static void RandomizeColors()
    {
        GameObject[] allObjects = FindObjectsOfType(typeof(GameObject)) as GameObject[];
        
        foreach (GameObject obj in allObjects)
        {
            Renderer objRenderer = obj.GetComponent<Renderer>();

            if (objRenderer != null)
            {
                objRenderer.material.color = new Color(Random.value, Random.value, Random.value, 1.0f);
            }   
        }
    }
}
