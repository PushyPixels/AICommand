using UnityEngine;
using UnityEditor;

public static class TerrainGenerator
{
    [MenuItem("Edit/Do Task")]
    public static void GenerateTerrain()
    {
        TerrainData terrainData = new TerrainData();

        // Set terrain size
        terrainData.size = new Vector3(1000, 300, 1000);

        // Set terrain heightmap resolution
        terrainData.heightmapResolution = 513;

        // Generate terrain
        float[,] heights = GenerateHeights(terrainData.heightmapResolution);
        terrainData.SetHeights(0, 0, heights);
        
        // Add the terrain to the scene
        GameObject terrainGO = Terrain.CreateTerrainGameObject(terrainData);
    }

    private static float[,] GenerateHeights(int resolution)
    {
        float[,] heights = new float[resolution, resolution];

        // Generate Perlin noise
        for (int x = 0; x < resolution; x++)
        {
            for (int y = 0; y < resolution; y++)
            {
                float xCoord = (float)x / resolution * 10f;
                float yCoord = (float)y / resolution * 10f;
                float height = Mathf.PerlinNoise(xCoord, yCoord);
                heights[x, y] = height;
            }
        }

        return heights;
    }
}
